const DEFAULT_SERVER_URL = "http://localhost:8787";
function getServerUrl(){return localStorage.getItem("defensaipn_server_url")||DEFAULT_SERVER_URL;}
function renderServerBadge(){
  const u=getServerUrl();const el=document.getElementById("serverUrlBadge");
  if(!el)return;try{el.textContent=new URL(u).host;}catch{el.textContent="local";}
}

// Page navigation
function showPage(name){
  const NEEDS_CASE=["import","documents","chat","drafts"];
  if(NEEDS_CASE.includes(name)&&!state.activeCaseId){openNewCaseModal();return;}
  document.querySelectorAll(".page").forEach(p=>p.classList.remove("active"));
  document.querySelectorAll(".nav-item").forEach(n=>n.classList.remove("active"));
  const pg=document.getElementById("page-"+name);
  if(pg)pg.classList.add("active");
  const nav=document.getElementById("nav-"+name);
  if(nav)nav.classList.add("active");
  if(name==="documents")renderDocuments();
  if(name==="chat")renderChat();
  if(name==="drafts")renderDrafts();
  if(name==="casos")renderCaseGrid();
  if(name==="import"){renderQueue();const sub=document.getElementById("importCaseSub");if(sub)sub.textContent=currentCaseName();}
  if(name==="ajustes"){renderStorage();checkHealth().catch(()=>{});resolveModel().catch(()=>{});}
}
function currentCaseName(){return state._activeCase?state._activeCase.name:"Sin expediente";}

// Modal
function openNewCaseModal(){document.getElementById("newCaseModal").style.display="";}
function closeNewCaseModal(){document.getElementById("newCaseModal").style.display="none";}
async function createCaseFromModal(){
  const name=document.getElementById("newCaseName").value.trim();

  document.getElementById("caseName").value=name;
  document.getElementById("caseType").value=document.getElementById("newCaseType").value;

}

// Render case grid
async function renderCaseGrid(){
  const cases=await all("cases");const docs=await all("documents");
  const msgs=await all("messages");
  const grid=document.getElementById("caseGrid");if(!grid)return;
  let html='<div class="case-card add-case-card" onclick="openNewCaseModal()"><div style="font-size:2rem;margin-bottom:8px">➕</div><div style="font-weight:700;font-size:.9rem">Nuevo expediente</div><div style="font-size:.75rem;color:var(--mut);margin-top:4px">Crear un nuevo caso</div></div>';
  for(const c of cases){
    const nd=docs.filter(d=>d.caseId===c.id).length;
    const nm=msgs.filter(m=>m.caseId===c.id).length;
    const isCur=c.id===state.activeCaseId;
    const icon=c.type&&c.type.includes("ARCO")?"🗂":c.type&&c.type.includes("Queja")?"⚠️":c.type&&c.type.includes("Denuncia")?"📢":"📁";
    html+=`<div class="case-card${isCur?" current":""}" onclick="selectCaseAndGo('${c.id}')">
      ${isCur?'<div class="cc-active-badge">ACTIVO</div>':""}
      <div class="cc-icon">${icon}</div>
      <div class="cc-name">${esc(c.name)}</div>
      <div class="cc-type">${esc(c.type||"")}</div>
      <div class="cc-stats"><span>📄 ${nd} docs</span><span>💬 ${nm} chats</span></div>
    </div>`;
  }
  grid.innerHTML=html;
}
async function selectCaseAndGo(id){
  await selectCase(id);
  showPage("chat");
}

const _origRenderDrafts=window.renderDrafts;
async function renderDrafts(){
  const drafts=(await draftsOfActive()).sort((a,b)=>b.updatedAt.localeCompare(a.updatedAt));
  const list=document.getElementById("draftList");
  if(list)list.innerHTML=drafts.map(d=>`<div class="draft-item${d.id===state.selectedDraftId?" sel":""}" onclick="selectDraft('${d.id}')"><div style="font-weight:600;font-size:.85rem">${esc(d.title)}</div><div style="font-size:.7rem;color:var(--mut);margin-top:3px">${esc(new Date(d.updatedAt).toLocaleString())}</div><div class="btn-row" style="margin-top:8px"><button class="btn btn-danger btn-sm" onclick="event.stopPropagation();deleteDraft('${d.id}')">🗑</button></div></div>`).join("")||"<div style='color:var(--mut);font-size:.83rem;padding:10px 0'>Sin borradores guardados.</div>";
  const sel=state.selectedDraftId?await get("drafts",state.selectedDraftId):null;
  const meta=document.getElementById("draftMeta");
  const ed=document.getElementById("draftEditor");
  if(meta)meta.textContent=sel?sel.title:"Selecciona un borrador de la lista";
  if(ed)ed.value=sel?.content||"";
}

// Chat render override (remove citation audit noise from display)
const _origRenderChat=window.renderChat;
async function renderChat(){
  const msgs=(await messagesOfActive()).sort((a,b)=>a.createdAt.localeCompare(b.createdAt));
  const box=document.getElementById("chatBox");if(!box)return;
  if(!msgs.length){box.innerHTML="<div style='color:var(--mut);text-align:center;margin-top:40px;font-size:.9rem'>No hay mensajes. Escribe tu primera pregunta arriba.</div>";return;}
  box.innerHTML=msgs.map(m=>{
    const text=(m.content||"").replace(/\n---\nVALIDACIÓN LOCAL.*$/s,"").trim();
    const isUser=m.role==="user";
    const time=new Date(m.createdAt).toLocaleTimeString("es-MX",{hour:"2-digit",minute:"2-digit"});
    return `<div class="msg ${m.role}"><div class="msg-meta">${isUser?"Tú":("IA"+(m.meta?.model?" · "+esc(m.meta.model):""))} · ${time}</div>${esc(text)}</div>`;
  }).join("");
  box.scrollTop=box.scrollHeight;
}

// configServerUrl
function configServerUrl(){
  const cur=localStorage.getItem("defensaipn_server_url")||DEFAULT_SERVER_URL;
  const url=prompt("URL del servidor Express:\nPor defecto: "+DEFAULT_SERVER_URL+"\nDeja vacío para restablecer el valor por defecto.\nEjemplo remoto: http://192.168.1.10:8787",cur);
  if(url===null)return;
  const t=url.trim();
  if(t&&t!==DEFAULT_SERVER_URL)localStorage.setItem("defensaipn_server_url",t);
  else localStorage.removeItem("defensaipn_server_url");
  renderServerBadge();
  checkHealth().catch(()=>{});
}
